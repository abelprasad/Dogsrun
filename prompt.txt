### FIX 2: Create API route for Mark as Urgent

Create src/app/api/dogs/set-urgent/route.ts

- Accepts POST { dog_id }
- Uses Supabase service role key (process.env.SUPABASE_SERVICE_ROLE_KEY)
- Updates dogs table: SET status = 'urgent' WHERE id = dog_id
- Returns NextResponse.json({ success: true })
- Returns 400 if dog_id missing, 500 if update fails

import { createClient } from '@supabase/supabase-js'

const supabaseAdmin = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
)

const { data: alertsData } = await supabaseAdmin
  .from('alerts')
